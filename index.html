const express = require('express'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); // REEMPLAZA ACÁ CON TU API KEY DE GEMINI (CONSEGUILA EN GOOGLE AI STUDIO) const GEMINI_API_KEY = "TU_API_KEY_AQUI"; // 1. ENDPOINT PARA LA RESEÑA AUTOMÁTICA CON HUMOR app.post('/api/narrative', async (req, res) => { const { weatherData, locationName } = req.body; if (!weatherData) return res.status(400).json({ error: 'Faltan datos del clima' }); const prompt = { contents: [{ parts: [{ text: `Actúa como un meteorólogo e influencer argentino/rioplatense con un humor ácido, irónico pero súper amigable. Analizá estos datos del clima actuales para la ciudad de "${locationName}": - Temperatura actual: ${weatherData.temp}°C - Sensación térmica: ${weatherData.sensacion}°C - Humedad: ${weatherData.humedad}% - Lluvia hoy: ${weatherData.lluvia} mm - Condición general: ${weatherData.condicion} Escribí una reseña cortita (máximo 3 líneas), directa, sin tecnicismos aburridos, usando modismos rioplatenses ("che", "viste", "olvidate", "fiero", "re pesado", "saquito"). Que sea graciosa y acorde estrictamente a si hace frío, calor, si está húmedo o si se viene la tormenta.` }] }] }; try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(prompt) }); const data = await response.json(); const text = data.candidates[0].content.parts[0].text; res.json({ result: text.trim() }); } catch (error) { console.error(error); res.json({ result: `Y... qué querés que te diga, el servidor tiró error. Pero hace ${weatherData.temp}°C, arreglate con eso.` }); } }); // 2. ENDPOINT PARA EL CHAT INTERACTIVO app.post('/api/chat', async (req, res) => { const { message, weatherData, locationName, history } = req.body; // Formateamos el historial para mantener la conversación coherente const formattedHistory = (history || []).map(msg => ({ role: msg.role === 'user' ? 'user' : 'model', parts: [{ text: msg.content }] })); const systemInstruction = `Sos el experto en clima interactivo de "Clima para Dummies". Hablás en perfecto español rioplatense (de forma casual, divertida, usando "che", "viste", "ni lo pienses"). Tu contexto actual es la ciudad de "${locationName}". Datos del clima: Temperatura ${weatherData.temp}°C, Sensación Térmica ${weatherData.sensacion}°C, Humedad ${weatherData.humedad}%, Lluvia de hoy ${weatherData.lluvia} mm, Condición: ${weatherData.condicion}. Respondé la pregunta del usuario usando estos datos con mucha onda, humor y practicidad para la vida cotidiana.`; try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [ ...formattedHistory, { role: 'user', parts: [{ text: message }] } ], systemInstruction: { parts: [{ text: systemInstruction }] } }) }); const data = await response.json(); const text = data.candidates[0].content.parts[0].text; res.json({ result: text.trim() }); } catch (error) { console.error(error); res.json({ result: "Che, se me congeló la neurona artificial. Preguntame de nuevo en un toque." }); } }); app.listen(3000, () => console.log('🚀 Servidor de Clima para Dummies corriendo en http://localhost:3000'));
